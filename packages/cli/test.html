🔧 Deduplicate /api/:id/envelope across 2 components @tanstack/react-query
2 different components (Li, Yt) each independently fetch "/api/:id/envelope". TanStack Query automatically deduplicates requests with the same query key, so extracting a shared hook means only 1 network request fires regardless of how many components call it.
hooks/useEnvelope.ts 
// ✅ FIX: Extract shared query hook — called from 2 components
// Before: 7 duplicate requests to /api/:id/envelope
// After:  1 request, shared via TanStack Query cache
// Saves:  6 requests, ~1KB bandwidth

import { useQuery } from '@tanstack/react-query';

// Shared hook — use this in all components that need this data
export function useEnvelope() {
  return useQuery({
    queryKey: ['envelope'],
    queryFn: async () => {
      const res = await fetch('/api/:id/envelope');
      if (!res.ok) throw new Error('Failed to fetch envelope');
      return res.json();
    },
    staleTime: 30 * 1000,  // 30 seconds — prevents refetch across components
    gcTime: 5 * 60 * 1000, // 5 minutes in cache
  });
}

🔧 Batch 7 individual envelope requests @tanstack/react-query
The Yt component fires 7 separate requests to "/api/:id/envelope", one per list item. This is the classic N+1 problem. The fix is either a batch endpoint (`?ids=1,2,3`) or `Promise.all` for parallel fetching. This eliminates 6 requests and saves ~1207ms.

// ✅ FIX: Replace 7 individual requests with 1 batch request
// Before: GET /api/:id/envelope × 7 (1610ms total)
// After:  GET /api/:id/envelope?ids=1,2,3... × 1 (~403ms)
// Saves:  6 requests, ~1207ms

import { useQuery } from '@tanstack/react-query';

// Option A: If your API supports batch endpoint
function useEnvelopeBatch(ids: string[]) {
  return useQuery({
    queryKey: ['envelope', 'batch', ids],
    queryFn: async () => {
      const res = await fetch(`/api/:id/envelope?ids=${ids.join(',')}`);
      if (!res.ok) throw new Error('Batch fetch failed');
      return res.json();
    },
    enabled: ids.length > 0,
  });
}

// Option B: If no batch endpoint exists, use Promise.all with dedup
function useEnvelopeByIds(ids: string[]) {
  return useQuery({
    queryKey: ['envelope', 'multi', ids],
    queryFn: () => Promise.all(
      ids.map(id =>
        fetch(`/api/:id/envelope/${id}`).then(r => r.json())
      )
    ),
    enabled: ids.length > 0,
    staleTime: 60 * 1000,
  });
}

// Usage in Yt:
function Yt({ itemIds }: { itemIds: string[] }) {
  const { data: envelope } = useEnvelopeBatch(itemIds);

  return (
    <ul>
      {envelope?.map(item => (
        <li key={item.id}>{item.name}</li>
      ))}
    </ul>
  );
}
